set(CMAKE_SYSTEM_NAME OwOS)
set(CMAKE_SYSTEM_PROCESSOR "i686")

set(TOOLCHAIN_PREFIX ${CMAKE_SOURCE_DIR}/toolchain-hosted/bin/i686-owos-)

set(CMAKE_SYSROOT ${CMAKE_SOURCE_DIR}/bin/sysroot)
set(CMAKE_STAGING_PREFIX ${CMAKE_SOURCE_DIR}/bin/sysroot)
set(CMAKE_INSTALL_PREFIX ${CMAKE_SOURCE_DIR}/bin/sysroot)
set(CMAKE_INSTALL_BINDIR sysroot/bin)
set(CMAKE_INSTALL_LIBDIR sysroot/lib)
set(CMAKE_INSTALL_INCLUDEDIR sysroot/usr/include)
set(CMAKE_INSTALL_DATAROOTDIR sysroot/share)

set(CMAKE_C_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_CXX_COMPILER ${TOOLCHAIN_PREFIX}g++)
set(CMAKE_ASM_COMPILER ${TOOLCHAIN_PREFIX}gcc)
set(CMAKE_LINKER ${TOOLCHAIN_PREFIX}ld)
set(CMAKE_AR ${TOOLCHAIN_PREFIX}gcc-ar)
set(CMAKE_NM ${TOOLCHAIN_PREFIX}gcc-nm)
set(CMAKE_OBJCOPY ${TOOLCHAIN_PREFIX}objcopy)
set(CMAKE_RANLIB ${TOOLCHAIN_PREFIX}gcc-ranlib)
set(CMAKE_STRIP ${TOOLCHAIN_PREFIX}strip)

set(CMAKE_C_COMPILER_WORKS TRUE)
set(CMAKE_CXX_COMPILER_WORKS TRUE)

set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_PACKAGE BOTH)
